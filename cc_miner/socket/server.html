 59    async def register(self, websocket: WebSocketServerProtocol, _id: int) -> None:
 60        """Register a websocket connection.
 61
 62        Args:
 63            websocket (WebSocketServerProtocol): The websocket connection.
 64            _id (int): The unique id of the turtle.

 66        turtle = self.mining_type(_id, websocket)

 81    async def handler(self, websocket: WebSocketServerProtocol, path: str) -> None:
 82        """The handler for the server.
 83
 84        Handles the websocket connection before a connection is properly established.
 85
 86        Args:
 87            websocket (WebSocketServerProtocol): The websocket connection.
 88            path (str): The path to the websocket.
 89        """
 90        message = await websocket.recv()
 91        logger.debug("Got message: %s", message)
 92
 93        try:
 94            event = json.loads(message)
 95        except json.JSONDecodeError:
 96            logger.error("Not a valid JSON message: %s", message)
 97            raise await self.error(websocket, "Invalid JSON message")
 98
 99        if "type" not in event:
100            logger.error("No type in message: %s", message)
101            raise await self.error(websocket, "'type' key not in message")
102
103        for potential_type in [
104            DataMessage,
105            ErrorMessage,
106            CommandResponse,
107            RegisterMessage,
108        ]:
109            try:
110                event = potential_type.parse_obj(event)
111            except ValidationError:
112                continue
113            else:
114                break
115        else:
116            raise await self.error(websocket, f"Could not parse: {event}")
117
118        if type(event) is RegisterMessage:
119            logger.info("Got new registration from #%s", event.id)
120            await self.register(websocket, event.id)
121        else:
122            raise await self.error(websocket, f"Unexpected event type: {event.json()}")
123
124    async def output_statuses(self) -> None:
125        """Output the status of all active turtles."""
126        logging.getLogger("cc_miner").setLevel(logging.WARNING)
127        while True:
128            system("clear")  # clear the screen
129            if self.clients:
130                print(f"{len(self.clients)} active turtles:\n\n")
131                for turtle in self.clients:
132                    output = await turtle.get_status()
133                    print(f"Turtle #{turtle.uid}:\n{output}\n\n")
134            else:
135                print("No active turtles...")
136            await asyncio.sleep(0.1)

139        """Start the server."""

144        asyncio.get_event_loop().run_until_complete(self.output_statuses())

146        try:
147            asyncio.get_event_loop().run_forever()

149            logger.info("Stopping server")
150            asyncio.get_event_loop().stop()
151            asyncio.get_event_loop().run_forever()
152            asyncio.get_event_loop().close()

482    """A turtle that can mine a stripmine."""
483
484    branch_spacing: int = 3
485    """blocks to leave between each branch"""
486    branch_length: int = 47
487    """number of blocks to mine in each branch"""
488    branch_pair_count: int = 8
489    """total number of pairs of branches"""
490    prerun_fuel_check: bool = True
491    """check if enough fuel before mining"""
492    do_place_torches: bool = True
493    """whether to place torches in the stripmine"""
494    torch_light: int = 12
495    """amount of blocks that torch light travels"""
496    current_light_level: int = 0
497    """current light level on the turtle"""

499    async def place_torch(self) -> None:
500        """Try to place a torch above the turtle."""
501        try:
502            await self.inventory_select("torch")
503        except InventoryException:
504            self._logger.error("Ran out of torches.")
505            raise
506        else:
507            await self.place_block(Direction.UP)
508
509    @overrides
510    async def get_status(self) -> str:
511        """Create a status update string."""
512        output = await super().get_status()
513        output += f"Light Level:     {self.current_light_level}\n"
514        return output

516    @overrides
517    async def start(self) -> None:
518        """The main turtle process."""
519        if self.prerun_fuel_check:
520            required_fuel: int = (
521                (
522                    self.branch_spacing + 1
523                )  # spacing between branches, plus 1 for the connection between branches
524                + (
525                    self.branch_length * 4 + 1
526                )  # length of mining each branch and going back to main
527            ) * self.branch_pair_count
528            current_fuel = await self.get_fuel()
529            if current_fuel < required_fuel:
530                raise HaltException(
531                    f"Not enough fuel to complete trip. Need {required_fuel - current_fuel} more."
532                )
533
534        for _ in range(self.branch_pair_count):
535            # continue main branch
536            for _ in range(self.branch_spacing + 1):
537                await self.dig_move(Direction.FORWARD)
538                await self.dig(Direction.UP)
539
540            # update home location current point on main branch
541            self._home_location = self.position.location
542
543            # mine left branch
544            await self.turn_left()
545            for _ in range(2):
546                for _ in range(self.branch_length):
547                    await self.dig_move(Direction.FORWARD)
548                    await self.dig(Direction.UP)
549
550                # go back to main branch
551                await self.turn_right()
552                await self.turn_right()
553
554                first_torch = True
555                self.current_light_level = self.torch_light
556                for branch_position in range(self.branch_length):
557                    # place torches if necessary
558                    target_light: int = 0 if first_torch else -(self.torch_light + 1)
559
560                    if self.do_place_torches and self.current_light_level <= target_light:
561                        if first_torch:
562                            first_torch = False
563
564                        try:
565                            await self.place_torch()
566                        except InventoryException:
567                            # place failed
568                            self.do_place_torches = False
569                        else:
570                            # place success
571                            self.current_light_level = self.torch_light
572
573                    # we don't need to dig move because we already mined the blocks
574                    await self.move(Direction.FORWARD)
575                    self.current_light_level -= 1  # decrease light level because we moved
576
577                    # at end of branch if there's not enough light, slap a torch down
578                    if (
579                        branch_position == (self.branch_length - 2)
580                        and self.do_place_torches
581                        and self.current_light_level <= -1
582                    ):
583                        try:
584                            await self.place_torch()
585                        except InventoryException:
586                            # place failed
587                            self.do_place_torches = False
588                        else:
589                            # place success
590                            self.current_light_level = self.torch_light
591
592            # face forward again to prepare for next branch pair
593            await self.turn_right()
